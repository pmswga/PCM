class PDictionary
{
private $name;
private $data;
public function construct($name)
{
$this->name = $name;
}
public function addEntry($key, $value)
{
$this->data[$key] = $value;
}
public function removeEntry($key)
{
unset($this->data[$key]);
}
public function getData()
{
return $this->data;
}
public function getName()
{
return $this->name;
}
public function sort()
{
return ksort($this->data);
}
}
